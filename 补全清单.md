```python
class RedisCoreMemoryStore(CoreMemoryStoreInterface):
    """
    使用 Redis HASH + 版本号实现 CAS。
    key = "cm:{user_id}:{block}"
    field = "data" / "version"
    """
    
    async def get_with_version(self, key: str) -> tuple[dict, int]:
        data = await self.redis.hget(f"cm:{key}", "data")
        version = await self.redis.hget(f"cm:{key}", "version") or 0
        return (json.loads(data) if data else {}), int(version)
    
    async def cas_upsert(self, key: str, value: dict, expected_version: int) -> bool:
        lua = """
        local current = redis.call('HGET', KEYS[1], 'version')
        if current and tonumber(current) ~= tonumber(ARGV[1]) then
            return 0
        end
        redis.call('HSET', KEYS[1], 'data', ARGV[2])
        redis.call('HINCRBY', KEYS[1], 'version', 1)
        return 1
        """
        result = await self.redis.eval(lua, 1, f"cm:{key}", expected_version, json.dumps(value))
        return bool(result)
    
    async def force_upsert(self, key: str, value: dict) -> None:
        pipe = self.redis.pipeline()
        pipe.hset(f"cm:{key}", "data", json.dumps(value))
        pipe.hincrby(f"cm:{key}", "version", 1)
        await pipe.execute()
    
    async def get_core_memory(self, user_id: str) -> CoreMemory:
        # 加载四个区块并组装
        pass
```

```python
# services/llm.py
from openai import AsyncOpenAI

class LLMInterface:
    def __init__(self, api_key: str, model: str = "gpt-4"):
        self.client = AsyncOpenAI(api_key=api_key)
        self.model = model
    
    async def generate(self, prompt: str) -> str:
        response = await self.client.chat.completions.create(
            model=self.model,
            messages=[{"role": "user", "content": prompt}]
        )
        return response.choices[0].message.content
    
    async def generate_json(self, prompt: str) -> dict:
        response = await self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "user", "content": prompt},
                {"role": "user", "content": "请以JSON格式返回。"}
            ],
            response_format={"type": "json_object"}
        )
        return json.loads(response.choices[0].message.content)
```

```python
class QdrantVectorDB(VectorDBInterface):
    def __init__(self, url: str, api_key: str):
        self.client = qdrant_client.QdrantClient(url=url, api_key=api_key)
    
    async def insert(self, entry: VectorEntry) -> None:
        vector = await self._embed(entry.content)
        self.client.upsert(
            collection_name=entry.namespace,
            points=[{"id": entry.id, "vector": vector, "payload": entry.model_dump()}]
        )
    
    async def search(self, query_embedding: list[float], namespace: str, top_k: int) -> list[VectorEntry]:
        results = self.client.search(
            collection_name=namespace,
            query_vector=query_embedding,
            limit=top_k
        )
        return [VectorEntry(**r.payload) for r in results]
```

```python
from fastapi import FastAPI
import uvicorn

app = FastAPI()

@app.post("/chat")
async def chat(user_id: str, session_id: str, message: str):
    # 1. 构建 prompt
    prompt = await app.state.soul_engine.build_prompt(
        user_id, session_id, message
    )
    
    # 2. LLM 推理
    result = await app.state.soul_engine.think(prompt)
    
    # 3. 路由动作
    action_result = await app.state.action_router.route(
        result, 
        {"user_id": user_id, "session_id": session_id}
    )
    
    # 4. 发射对话结束事件（触发进化）
    await app.state.event_bus.emit("dialogue_ended", {
        "user_id": user_id,
        "session_id": session_id,
        "dialogue": [
            {"role": "user", "content": message},
            {"role": "assistant", "content": action_result.get("content", "")}
        ]
    })
    
    return action_result

# 启动
if __name__ == "__main__":
    uvicorn.run(app, port=8000)
```
